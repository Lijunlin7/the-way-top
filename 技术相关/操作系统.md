# 操作系统

以下是实现 `time.c` 的两个版本的代码，分别使用 **共享内存** 和 **管道** 作为 IPC 机制，用于测量命令行命令的执行时间。

---

###  版本 1：使用共享内存（Shared Memory）

```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/wait.h>
#include <sys/time.h>
#include <sys/shm.h>
#include <sys/ipc.h>

int main(int argc, char *argv[]) {
    if (argc < 2) {
        fprintf(stderr, "Usage: %s <command>\n", argv[0]);
        exit(1);
    }

    // 创建共享内存
    int shm_id = shmget(IPC_PRIVATE, sizeof(struct timeval), IPC_CREAT | 0666);
    if (shm_id == -1) {
        perror("shmget");
        exit(1);
    }

    // 附加共享内存
    struct timeval *start_time = (struct timeval *)shmat(shm_id, NULL, 0);
    if (start_time == (void *)-1) {
        perror("shmat");
        exit(1);
    }

    pid_t pid = fork();
    if (pid == -1) {
        perror("fork");
        exit(1);
    }

    if (pid == 0) { // 子进程
        gettimeofday(start_time, NULL); // 记录开始时间
        execvp(argv[1], &argv[1]);     // 执行命令
        perror("execvp");               // 如果execvp失败
        exit(1);
    } else {        // 父进程
        wait(NULL);  // 等待子进程结束

        struct timeval end_time;
        gettimeofday(&end_time, NULL);  // 记录结束时间

        // 计算时间差（秒 + 微秒）
        double elapsed = (end_time.tv_sec - start_time->tv_sec) + 
                         (end_time.tv_usec - start_time->tv_usec) / 1000000.0;

        printf("Elapsed time: %.5f seconds\n", elapsed);

        // 释放共享内存
        shmdt(start_time);
        shmctl(shm_id, IPC_RMID, NULL);
    }

    return 0;
}
```

### **版本 2：使用管道（Pipe）**
```c
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <sys/wait.h>
#include <sys/time.h>

int main(int argc, char *argv[]) {
    if (argc < 2) {
        fprintf(stderr, "Usage: %s <command>\n", argv[0]);
        exit(1);
    }

    int pipe_fd[2];
    if (pipe(pipe_fd) == -1) {
        perror("pipe");
        exit(1);
    }

    pid_t pid = fork();
    if (pid == -1) {
        perror("fork");
        exit(1);
    }

    if (pid == 0) { // 子进程
        close(pipe_fd[0]); // 关闭读端

        struct timeval start_time;
        gettimeofday(&start_time, NULL); // 记录开始时间

        // 将时间写入管道
        write(pipe_fd[1], &start_time, sizeof(start_time));
        close(pipe_fd[1]);

        execvp(argv[1], &argv[1]); // 执行命令
        perror("execvp");          // 如果execvp失败
        exit(1);
    } else {        // 父进程
        close(pipe_fd[1]); // 关闭写端

        struct timeval start_time;
        read(pipe_fd[0], &start_time, sizeof(start_time)); // 读取开始时间
        close(pipe_fd[0]);

        wait(NULL); // 等待子进程结束

        struct timeval end_time;
        gettimeofday(&end_time, NULL); // 记录结束时间

        // 计算时间差
        double elapsed = (end_time.tv_sec - start_time.tv_sec) + 
                         (end_time.tv_usec - start_time.tv_usec) / 1000000.0;

        printf("Elapsed time: %.5f seconds\n", elapsed);
    }

    return 0;
}
```

---

### **关键点说明**
1. **共享内存版本**：
   - 使用 `shmget` 创建共享内存，`shmat` 附加到进程地址空间。
   - 子进程写入开始时间，父进程读取并计算时间差。
   - 最后通过 `shmdt` 和 `shmctl` 释放共享内存。

2. **管道版本**：
   - 使用 `pipe` 创建匿名管道。
   - 子进程通过 `write` 写入开始时间，父进程通过 `read` 读取。
   - 管道会自动处理进程间同步。

3. **时间测量**：
   - `gettimeofday` 获取时间戳（秒和微秒）。
   - 时间差 = `(end_sec - start_sec) + (end_usec - start_usec) / 1e6`。

4. **错误处理**：
   - 检查 `fork`、`execvp`、`shmget` 等系统调用的返回值。
